Отобрать список пациентов, у которых сегодня день рождения. Вывести ФИО пациента

SELECT FULLNAME // выбрать полное имя пациента
FROM CLIENTS // из таблицы "пациенты"
WHERE MONTH(BDATE) = MONTH(CURRENT_DATE) // проверяем, что месяц даты рождения совпадает с текущим месяцем
AND DAY(BDATE) = DAY(CURRENT_DATE); // проверяем, что день даты рождения совпадает с текущим днем.
