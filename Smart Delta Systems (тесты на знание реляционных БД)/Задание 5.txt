Вычислить вторую максимальную дату рождения в таблице пациентов

SELECT MAX(BDATE) as Second_Max_BDATE
// Создаём запрос, который находит максимальную дату рождения (MAX(BDATE)) из всех клиентов в таблице CLIENTS. 
// Это будет самая поздняя дата рождения среди всех клиентов.

FROM CLIENTS
// Говорим "откуда" это вытянуть - из таблицы CLIENTS.

WHERE BDATE < (SELECT MAX(BDATE) FROM CLIENTS);
// Делаем условие, по которому фильтруем даты, оставляя только те, 
// которые меньше максимальной даты рождения, т.е. меньше конструкции выше.
