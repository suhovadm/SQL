Вычислить вторую максимальную дату рождения в таблице пациентов, 
не используя агрегатных функций (по условию могут быть пациенты с одинаковой датой рождения)

SELECT DISTINCT BDATE
// Выбираем уникальные даты из столбца BDATE, чтобы избежать дубликатов.

FROM CLIENTS
// Говорим откуда - из таблицы CLIENTS.

ORDER BY BDATE DESC
// Включаем сортировку в порядке убывания.
// Таким образом наиболее поздние даты идут первыми.

OFFSET 1 ROW
// Пропускаем первую строку в результате выдачи,
// это будет самая последняя дата рождения.

FETCH NEXT 1 ROW ONLY;
// Извлекаем только одну строку после пропуска.
